About FOREX QUANT

FOREX QUANT — це дослідницький проєкт машинного навчання для аналізу та прогнозування валютних ринків. Система поєднує класичний технічний аналіз, ML-моделі та AI-аналітику для генерації торгових сигналів.

Проєкт пройшов шлях від базового RSI/MACD аналізу до повноцінного AI-пайплайну з walk-forward валідацією, meta-labeling та інтеграцією великих мовних моделей через Groq API.

Автоматичні звіти доставляються через Telegram Bot щоденно — з прогнозами, ймовірностями та інституційним аналізом.

# Feature Engineering Pipeline
def build_features(df):
    # Technical indicators
    df['rsi_14'] = compute_rsi(df, 14)
    df['macd'], df['signal'] = compute_macd(df)
    df['bb_upper'], df['bb_lower'] = bollinger(df, 20)
    df['atr_14'] = atr(df, 14)

    # Market regime features
    df['trend'] = detect_regime(df['close'])
    df['volatility'] = df['close'].rolling(20).std()

    return df.dropna()
ai_prompt.py PYTHON 
# AI Analysis via Groq API
def ai_analyze(signal_data):
    prompt = f"""
    You are a Forex analyst.
    Pair: {signal_data['pair']}
    BUY prob: {signal_data['buy_prob']:.1%}
    SELL prob: {signal_data['sell_prob']:.1%}
    Regime: {signal_data['regime']}

    Provide: institutional bias,
    key levels, trade recommendation.
    """
    response = groq.chat(prompt)
    return response.content
model.py PYTHON 
# Stacking Ensemble Model
from sklearn.ensemble import StackingClassifier

estimators = [
    ('rf',  RandomForestClassifier(200)),
    ('xgb', XGBClassifier(300)),
    ('lgb', LGBMClassifier(200)),
]

model = StackingClassifier(
    estimators=estimators,
    final_estimator=LogisticRegression(),
    cv=5, passthrough=True
)

# Auto Telegram Report
async def send_report(signal, ai_text):
    emoji = "🟢" if signal['action']=='BUY' else "🔴"
    msg = (
        f"{emoji} **{signal['pair']}**\n"
        f"Signal: {signal['action']}\n"
        f"BUY:  {signal['buy_prob']:.1%}\n"
        f"SELL: {signal['sell_prob']:.1%}\n\n"
        f"🤖 AI: {ai_text}"
    )
    await bot.send_message(CHAT_ID, msg)
